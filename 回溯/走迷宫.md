#include<iostream>
#include<vector>
using namespace std;

const int row=5;
const int col=5;
int maze[row][col]={
    {1,0,1,1,1},
    {1,1,1,0,1},
    {0,0,0,1,1},
    {1,1,0,1,0},
    {1,1,1,1,1}
};
vector<pair<int,int>> d={{-1,0},{0,1},{1,0},{0,-1}};

bool isValid(int r,int c){
    return (r>=0&&r<row&&c>=0&&c<col&&maze[r][c]==1);
}
bool dfs(int r,int c,vector<pair<int,int>>& path){
    if(r==row-1&&c==col-1){
        path.push_back({r,c});
        return true;
    }
    maze[r][c]=0;
    path.push_back({r,c});
    for(int i=0;i<4;i++){
        int newL=r+d[i].first;
        int newC=c+d[i].second;
        if(isValid(newL,newC)&&dfs(newL,newC,path))return true;
    }
    path.pop_back();
    return false;
}
int main(){
    vector<pair<int,int>> path;
    if(dfs(0,0,path)){
        for(auto p:path)cout<<"("<<p.first<<","<<p.second<<")"<<endl;
    }else cout<<"没有路径"<<endl;
    return 0;
}
